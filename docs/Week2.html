0.6.1 Density

#
dens <- function(N, data = net, directed = TRUE) {

    if (directed == TRUE) {
        pos_tie <- (N * (N - 1))
    } else {
        pos_tie <- (N * (N - 1))/2

    }

    obs_tie <- sum(sna::dyad.census(data)[1:2])
    dens <- obs_tie/pos_tie

    return(dens)
}
dens(10, small_net, directed = TRUE)
#> [1] 0.1444444
